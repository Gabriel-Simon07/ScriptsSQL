select 
  	aluno.primeiro_nome,
  	aluno.ultimo_nome,
  	count(aluno_curso.curso_id) numero_cursos
from
  	aluno 
join 
  	aluno_curso 
on 
  	aluno_curso.aluno_id = aluno.id
group by 
  	1,2
order by 
	numero_cursos 
desc;


select 
 	curso.nome, 	
  	count(aluno_curso.aluno_id) numero_alunos 
from 
 	curso
join 
 	aluno_curso
on
 	aluno_curso.curso_id = curso.id
group by 
	1
order by 
	numero_alunos 
desc;
