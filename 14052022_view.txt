create 
view vw_cursos_por_categorias 
as
select
	categoria
from
	(
	select
		categoria.nome as categoria,
		count(curso.id) as numero_cursos
	from
		categoria
	join curso on
		curso.categoria_id = categoria.id
	group by
		categoria
) as categorias_cursos
where
	numero_cursos >= 3;


create view 
	vw_cursos_programacao 
as select 
	nome 
from 
	curso 
where 
	categoria_id = 2;
